import numpy as np
import matplotlib.pyplot as plt

# Data
X = np.array([1200, 1500, 2000, 1800, 2500, 3000, 2200, 2700])
Y = np.array([300000, 320000, 450000, 400000, 550000, 600000, 480000, 700000])
n = len(X)

# Calculating slope (m) and intercept (c)
sum_x = np.sum(X)
sum_y = np.sum(Y)
sum_xy = np.sum(X * Y)
sum_x2 = np.sum(X ** 2)

m = (n * sum_xy - sum_x * sum_y) / (n * sum_x2 - sum_x ** 2)
c = (sum_y - m * sum_x) / n

print(f"Slope (m): {m:.2f}")
print(f"Intercept (c): {c:.2f}")

# Predict at x = 1940
x_predict = 1940
y_predict = m * x_predict + c
print(f"Predicted price for 1940 sq.ft: ${y_predict:.2f}")

import numpy as np
import matplotlib.pyplot as plt

# Original Data (with outliers removed)
X = np.array([1200, 1500, 1800, 2000, 2200, 2500])  # Removed 2700, 3000
Y = np.array([300000, 320000, 400000, 450000, 480000, 550000])
n = len(X)

# Manual Linear Regression (y = m*x + c)
sum_x = np.sum(X)
sum_y = np.sum(Y)
sum_xy = np.sum(X * Y)
sum_x2 = np.sum(X ** 2)

m = (n * sum_xy - sum_x * sum_y) / (n * sum_x2 - sum_x ** 2)
c = (sum_y - m * sum_x) / n

print(f"Slope (m): {m:.2f}")
print(f"Intercept (c): {c:.2f}")

# Prediction at 1940 sq.ft
x_predict = 1940
y_predict = m * x_predict + c
print(f"Predicted price for 1940 sq.ft: ${y_predict:.2f}")

# Matrix method
X_mat = np.c_[np.ones(n), X]
Y_mat = Y.reshape(-1, 1)
theta = np.linalg.inv(X_mat.T @ X_mat) @ X_mat.T @ Y_mat
c2, m2 = theta.flatten()
y_pred_1940 = m2 * x_predict + c2

print(f"[Matrix] Slope (m): {m2:.2f}")
print(f"[Matrix] Intercept (c): {c2:.2f}")
print(f"[Matrix] Predicted price for 1940 sq.ft: ${y_pred_1940:.2f}")

# Predicted line values
Y_pred = m * X + c

# Plotting
plt.scatter(X, Y, color='blue', label='Actual data (filtered)')
plt.plot(X, Y_pred, color='red', label='Linear regression line (no outliers)')
plt.scatter(x_predict, y_predict, color='green', label='Prediction (1940 sq.ft)', zorder=5)
plt.xlabel("Size (sq.ft)")
plt.ylabel("Price ($)")
plt.title("House Price Prediction (Outliers Removed)")
plt.legend()
plt.grid(True)
plt.show()
